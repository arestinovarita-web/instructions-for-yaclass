    def __init__(self):

        self.minsize(900, 600)
        self.configure(bg=BG_MAIN)
        self._current_section = -1  # -1 = домашняя страница
        self._build_ui()
        self._show_home()

    def _build_ui(self):

        logo_lbl = tk.Label(
            header,
            text="ЯКласс",

            font=("Georgia", 22, "bold"),
            padx=20,
            cursor="hand2",

        logo_lbl.pack(side="left", pady=10)
        logo_lbl.bind("<Button-1>", lambda e: self._show_home())

        title_lbl = tk.Label(
            header,
            text="Методические рекомендации по организации самостоятельной работы обучающихся",

    def _on_home_hover(self, entering):
        if self._current_section == -1:
            return
        self.home_btn.configure(bg=BG_SIDEBAR_HOV if entering else BG_SIDEBAR)

    def _on_nav_hover(self, btn, idx, entering):
        if idx == self._current_section:
            return
        btn.configure(bg=BG_SIDEBAR_HOV if entering else BG_SIDEBAR)

    def _on_frame_configure(self, event):
        self.canvas.configure(scrollregion=self.canvas.bbox("all"))

    def _on_canvas_configure(self, event):
        self.canvas.itemconfig(self.canvas_window, width=event.width)

    def _on_mousewheel(self, event):
        if event.num == 4:
            self.canvas.yview_scroll(-1, "units")
        elif event.num == 5:
            self.canvas.yview_scroll(1, "units")
        else:
            self.canvas.yview_scroll(int(-1 * (event.delta / 120)), "units")

    def _update_nav_state(self):
        """Обновить подсветку навигации."""
        # Домашняя кнопка
        if self._current_section == -1:
            self.home_btn.configure(bg=BG_SIDEBAR_SEL, fg=FG_SIDEBAR_SEL, font=("Georgia", 10, "bold"))
        else:
            self.home_btn.configure(bg=BG_SIDEBAR, fg=FG_SIDEBAR, font=("Georgia", 10))
        # Разделы
        for i, btn in enumerate(self.nav_buttons):
            if i == self._current_section:
                btn.configure(bg=BG_SIDEBAR_SEL, fg=FG_SIDEBAR_SEL, font=("Georgia", 10, "bold"))
            else:
                btn.configure(bg=BG_SIDEBAR, fg=FG_SIDEBAR, font=("Georgia", 10))

    def _render_table(self, parent, table_data):
        cap_lbl = tk.Label(
            parent,
            text=table_data["caption"],
            bg=BG_CARD, fg=FG_MUTED,
            font=("Georgia", 10, "italic"),
            anchor="w",
        )
        cap_lbl.pack(fill="x", pady=(10, 4))

        tbl_frame = tk.Frame(parent, bg=DIVIDER, bd=1, relief="solid")
        tbl_frame.pack(fill="x", pady=(0, 6))

        headers = table_data["headers"]
        rows = table_data["rows"]
        num_cols = len(headers)

        col_weights = [1] * num_cols
        if num_cols == 3:
            col_weights = [2, 2, 3]
        elif num_cols == 2:
            col_weights = [2, 5]
        elif num_cols == 4:
            col_weights = [2, 2, 1, 2]

        for c in range(num_cols):
            tbl_frame.columnconfigure(c, weight=col_weights[c], uniform="col")

        for c, h in enumerate(headers):
            cell = tk.Frame(tbl_frame, bg=BG_TABLE_H)
            cell.grid(row=0, column=c, sticky="nsew", padx=1, pady=1)
            lbl = tk.Label(
                cell, text=h,
                bg=BG_TABLE_H, fg=FG_TABLE_H,
                font=("Georgia", 10, "bold"),
                padx=10, pady=7,
                anchor="w", justify="left",
                wraplength=220,
            )
            lbl.pack(fill="both", expand=True)

        for r, row in enumerate(rows):
            row_bg = BG_TABLE_R1 if r % 2 == 0 else BG_TABLE_R2
            for c, cell_text in enumerate(row):
                cell = tk.Frame(tbl_frame, bg=row_bg)
                cell.grid(row=r + 1, column=c, sticky="nsew", padx=1, pady=1)
                lbl = tk.Label(
                    cell, text=cell_text,
                    bg=row_bg, fg=FG_MAIN,
                    font=("Georgia", 10),
                    padx=10, pady=6,
                    anchor="nw", justify="left",
                    wraplength=220,
                )
                lbl.pack(fill="both", expand=True)
